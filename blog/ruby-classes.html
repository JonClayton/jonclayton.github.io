This is a post about creating a class. 


          class Pet
            def initialize (name)
              @name = name
            end
  
            def self.purpose
              puts "The purpose of a #{self} is love."
            end
  
            def eat
              puts "#{@name} eats"
            end
  
            def sleep
              puts "zzzzzz"
            end
  
          end

          class Dog < Pet
  
            def bark
              puts "#{@name} says 'Arf!'"
            end
    
            def eat
              puts "#{@name} drools and then eats"
            end  
  
          end

          Pet.purpose  
  
          kate = Pet.new("Kate")
          kate.eat
          kate.sleep
          #kate.bark # This creates an error and you'll need to comment it out to run Rover.
          #kate.purpose # #purpose is a class method, not an instance method, so kate, as an instance of the class, isn't a class and can't call it

          Dog.purpose

          rover = Dog.new("Rover")
          rover.eat
          rover.sleep
          rover.bark
          #rover.purpose # again, it won't work because rover is an instance, not a class.
